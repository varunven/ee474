49void schedulerSetup() {
50 // Initialize the timer

52 TCCR0B = (1 << CS02) ; // 1024 Prescaler
53 OCR0A = 117; // Full period of the clock every 500Hz/2ms
54 TIMSK0 = (1 << OCIE0A); // Enable Interrupts

69void schedulerUpdate() {
70 for (int i = 0; i < 10; i++) {
71 if (tasks[i] == nullptr) {
72 break;
73 }
74 if (states[i] == TaskState::READY) {
75 states[i] = TaskState::RUNNING;
76 tasks[i]();
77 if (states[i] != TaskState::SLEEPING) {
78 states[i] = TaskState::READY;
79 }
80 }
81 }

89void schedule_sync() {
90 while (sFlag == SchedulerState::PENDING) {
91 //no op

93 for(int i = 0; i < 10; i++) {
94 if (states[i] == TaskState::SLEEPING) {
95 sleeping[i] -= 2;
96 if (sleeping[i] < 2) {
97 states[i] = TaskState::READY;
98 }
99 }

101 sFlag = SchedulerState::PENDING;

123// Sets the output rate for TIMER4
124// Input: hertz - the hertz to drive TIMER4 at. If hertz is zero, the timer is disabled.

126int notes[] = {293, 329, 261, 130, 196};

132void enableTimer4() {
133 // Enable the clock
134 PRR1 &= ~(1 << PRTIM4);
135 // Enable the output pin
136 DDRH |= 1 << PH3;
137 // Clear the timer registers
138 TCCR4A = 0;
139 TCCR4B = 0;
140 // turn on CTC mode
141 TCCR4B |= (1 << WGM12);
142 // Set CS10 bit for a prescaler of 1
143 TCCR4B |= (1 << CS10);
144 // Enable output on PH3
145 TCCR4A |= 1 << COM4A0;
146 // Clear the timer counter
147 TCNT4 = 0;
148}
149
157void setTimer4Hertz(int hertz) {
158 if (hertz == 0) {
159 // Disable the timer output
160 TCCR4A &= ~(1 << COM4A0);
161 } else {
162 // Make sure the timer is enabled
163 TCCR4A |= 1 << COM4A0;
164 // Set the CTC value based on the hertz. Multiplying by two because the clock toggles, and so the value for the ORC4A needs to be toggled twice to be one cycle.
165 OCR4A = 16000000/(hertz * 2);
166 }
167}

203void seven_seg_updater() {
204 static uint8_t led = 0;
205 uint8_t value = (display % (sevenSegmentToDigit[1][led] * 10)) / sevenSegmentToDigit[1][led];
206 PORTF = (1 << PF7 | 1 << PF6 | 1 << PF5 | 1 << PF4) ^ sevenSegmentToDigit[0][led];
207 PORTK = digitToOutput[value];
208 led++;
209 led %= 4;
210 sleep_474(2);
