80void task_self_quit(){
81 uint8_t currentTask = get_running_task();
82 if (currentTask == 0xFF) {
83 Serial.println("Called task_self_quit with no running task");
84 return;
85 }
86 TCB* task = tasks[currentTask];
87 task->state = TaskState::DEAD;
88 remove_index_from_array(tasks, taskCount--, currentTask);

105void task_start(TCB* task) {
106 uint8_t deadTaskIndex = 0xFF;
107 for (uint8_t i = 0; i < deadTaskCount; i++) {
108 if (task == deadTasks[i]) {
109 deadTaskIndex = i;
110 }
111 }
112 if (deadTaskIndex == 0xFF) {
113 Serial.print("Task ");
114 Serial.write(task->desc);
115 Serial.println(" not found.");
116 } else {
117 remove_index_from_array(deadTasks, deadTaskCount--, deadTaskIndex);
118 }
119 task->state = TaskState::READY;
120 TCB* sync = tasks[taskCount - 1];
121 tasks[taskCount - 1] = task;
122 tasks[taskCount++] = sync;
123}

144void schedulerSetup() {
145 // Initialize the timer
146 TCCR0A = (1 << WGM01); // CTC mode
147 TCCR0B = (1 << CS02) ; // 1024 Prescaler
148 OCR0A = 117; // Full period of the clock every 500Hz/2ms
149 TIMSK0 = (1 << OCIE0A); // Enable Interrupts
150}

164void schedulerUpdate() {
165 for (int i = 0; i < 10; i++) {
166 TCB* task = tasks[i];
167 if (task == nullptr) {
168 break;
169 }
170 if (task->state == TaskState::READY) {
171 task->state = TaskState::RUNNING;
172 task->run();
173 if (task->state != TaskState::SLEEPING) {
174 task->state = TaskState::READY;
175 }
176 }
177 }
178}

185void schedule_sync() {
186 while (sFlag == SchedulerState::PENDING) {
187 //no op
188 }
189 for(int i = 0; i < 10; i++) {
190 TCB* task = tasks[i];
191 if (task == nullptr) {
192 continue;
193 }
194 if (task->state == TaskState::SLEEPING) {
195 task->sleep_time -= 2;
196 if (task->sleep_time < 2) {
197 task->state = TaskState::READY;
198 }
199 }
200 }
201 sFlag = SchedulerState::PENDING;
202}

231void enableTimer4() {
232 // Enable the clock
233 PRR1 &= ~(1 << PRTIM4);
234 // Enable the output pin
235 DDRH |= 1 << PH3;
236 // Clear the timer registers
237 TCCR4A = 0;
238 TCCR4B = 0;
239 // turn on CTC mode
240 TCCR4B |= (1 << WGM12);
241 // Set CS10 bit for a prescaler of 1
242 TCCR4B |= (1 << CS10);
243 // Enable output on PH3
244 TCCR4A |= 1 << COM4A0;
245 // Clear the timer counter
246 TCNT4 = 0;
247}
248
256void setTimer4Hertz(int hertz) {
257 if (hertz == 0) {
258 // Disable the timer output
259 TCCR4A &= ~(1 << COM4A0);
260 } else {
261 // Make sure the timer is enabled
262 TCCR4A |= 1 << COM4A0;
263 // Set the CTC value based on the hertz. Multiplying by two because the clock toggles, and so the value for the ORC4A needs to be toggled twice to be one cycle.
264 OCR4A = 16000000/(hertz * 2);
265 }
266}
