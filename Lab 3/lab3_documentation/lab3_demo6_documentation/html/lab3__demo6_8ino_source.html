78void task_self_quit(){
79 uint8_t currentTask = get_running_task();
80 if (currentTask == 0xFF) {
81 Serial.println("Called task_self_quit with no running task");
82 return;
83 }
84 TCB* task = tasks[currentTask];
85 task->state = TaskState::DEAD;
86 remove_index_from_array(tasks, taskCount--, currentTask);
87 if (deadTaskCount == 10) {
88 Serial.println("Quit task when no dead task slots available.");
89 return;
90 }
91 deadTasks[deadTaskCount++] = task;
92}

103void task_start(TCB* task) {
104 uint8_t deadTaskIndex = 0xFF;
105 for (uint8_t i = 0; i < deadTaskCount; i++) {
106 if (task == deadTasks[i]) {
107 deadTaskIndex = i;
108 }
109 }
110 if (deadTaskIndex == 0xFF) {
111 Serial.print("Task ");
112 Serial.write(task->desc);
113 Serial.println(" not found.");
114 } else {
115 remove_index_from_array(deadTasks, deadTaskCount--, deadTaskIndex);
116 }
117 task->state = TaskState::READY;
118 TCB* sync = tasks[taskCount - 1];
119 tasks[taskCount - 1] = task;
120 tasks[taskCount++] = sync;
121}

142void schedulerSetup() {
143 // Initialize the timer
144 TCCR0A = (1 << WGM01); // CTC mode
145 TCCR0B = (1 << CS02) ; // 1024 Prescaler
146 OCR0A = 117; // Full period of the clock every 500Hz/2ms
147 TIMSK0 = (1 << OCIE0A); // Enable Interrupts

162void schedulerUpdate() {
163 for (int i = 0; i < 10; i++) {
164 TCB* task = tasks[i];
165 if (task == nullptr) {
166 break;
167 }
168 if (task->state == TaskState::READY) {
169 task->state = TaskState::RUNNING;
170 task->run();
171 if (task->state != TaskState::SLEEPING) {
172 task->state = TaskState::READY;
173 }
174 }
175 }
176}

183void schedule_sync() {
184 while (sFlag == SchedulerState::PENDING) {
185 //no op
186 }
187 for(int i = 0; i < 10; i++) {
188 TCB* task = tasks[i];
189 if (task == nullptr) {
190 continue;
191 }
192 if (task->state == TaskState::SLEEPING) {
193 task->sleep_time -= 2;
194 if (task->sleep_time < 2) {
195 task->state = TaskState::READY;
196 }
197 }
198 }
199 sFlag = SchedulerState::PENDING;
200}

229void enableTimer4() {
230 // Enable the clock
231 PRR1 &= ~(1 << PRTIM4);
232 // Enable the output pin
233 DDRH |= 1 << PH3;
234 // Clear the timer registers
235 TCCR4A = 0;
236 TCCR4B = 0;
237 // turn on CTC mode
238 TCCR4B |= (1 << WGM12);
239 // Set CS10 bit for a prescaler of 1
240 TCCR4B |= (1 << CS10);
241 // Enable output on PH3
242 TCCR4A |= 1 << COM4A0;
243 // Clear the timer counter
244 TCNT4 = 0;
245}
246
254void setTimer4Hertz(int hertz) {
255 if (hertz == 0) {
256 // Disable the timer output
257 TCCR4A &= ~(1 << COM4A0);
258 } else {
259 // Make sure the timer is enabled
260 TCCR4A |= 1 << COM4A0;
261 // Set the CTC value based on the hertz. Multiplying by two because the clock toggles, and so the value for the ORC4A needs to be toggled twice to be one cycle.
262 OCR4A = 16000000/(hertz * 2);
263 }
264}

278void seven_seg_updater() {
279 static uint8_t led = 0;
280 if (quit_7seg) {
281 task_self_quit();
282 quit_7seg = false;
283 PORTF = 0;
284 PORTK = 0;
285 return;
286 }
287 uint8_t value = (display % (sevenSegmentToDigit[1][led] * 10)) / sevenSegmentToDigit[1][led];
288 PORTF = (1 << PF7 | 1 << PF6 | 1 << PF5 | 1 << PF4) ^ sevenSegmentToDigit[0][led];
289 PORTK = digitToOutput[value];
290 led++;
291 led %= 4;
292 sleep_474(2);
293}
