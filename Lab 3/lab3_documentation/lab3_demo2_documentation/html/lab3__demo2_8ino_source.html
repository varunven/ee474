73void schedulerUpdate() {
74 for (int i = 0; i < 10; i++) {
75 if (tasks[i] == nullptr) {
76 break;
77 }
78 if (states[i] == TaskState::READY) {
79 states[i] = TaskState::RUNNING;
80 tasks[i]();
81 if (states[i] != TaskState::SLEEPING) {
82 states[i] = TaskState::READY;
83 }
84 }
85 }
86}

94void schedule_sync() {
95 while (sFlag == SchedulerState::PENDING) {
96 //no op
97 }
98 for(int i = 0; i < 10; i++) {
99 if (states[i] == TaskState::SLEEPING) {
100 sleeping[i] -= 2;
101 if (sleeping[i] < 2) {
102 states[i] = TaskState::READY;
103 }
104 }
105 }
106 sFlag = SchedulerState::PENDING;
107}

137void enableTimer4() {
138 // Enable the clock
139 PRR1 &= ~(1 << PRTIM4);
140 // Enable the output pin
141 DDRH |= 1 << PH3;
142 // Clear the timer registers
143 TCCR4A = 0;
144 TCCR4B = 0;
145 // turn on CTC mode
146 TCCR4B |= (1 << WGM12);
147 // Set CS10 bit for a prescaler of 1
148 TCCR4B |= (1 << CS10);
149 // Enable output on PH3
150 TCCR4A |= 1 << COM4A0;
151 // Clear the timer counter
152 TCNT4 = 0;
153}

163void setTimer4Hertz(int hertz) {
164 if (hertz == 0) {
165 // Disable the timer output
166 TCCR4A &= ~(1 << COM4A0);
167 } else {
168 // Make sure the timer is enabled
169 TCCR4A |= 1 << COM4A0;
170 // Set the CTC value based on the hertz. Multiplying by two because the clock toggles, and so the value for the ORC4A needs to be toggled twice to be one cycle.
171 OCR4A = 16000000/(hertz * 2);
172 }
173}
